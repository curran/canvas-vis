    Red Canvas

  • ¶

    An example of

    • Creating a Component
    • Defining its paint method
    • Binding it to a Canvas
    require ['cv/Component', 'cv/bindToCanvas'], (Component, bindToCanvas) ->
  component = new Component
  
    paint: (ctx, bounds) ->

      ctx.fillStyle = 'red'
      ctx.fillRect(
        bounds.x,
        bounds.y,
        bounds.w,
        bounds.h
      )

  return bindToCanvas 'redCanvas', component
